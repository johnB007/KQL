//devices in MDE that are older than builds starting with 18
DeviceInfo 
| where OSPlatform == "Windows10"
| where toint(substring(OSBuild,0,2)) < 18
| summarize arg_max(Timestamp,*) by  DeviceName

//All Devices
DeviceInfo 
| where OSPlatform != "Windows10"
| where toint(substring(OSBuild,0,2)) != 1
| summarize arg_max(Timestamp,*) by  DeviceName
