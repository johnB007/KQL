DeviceLogonEvents 
| where IsLocalAdmin == 1 
| where AccountName !startswith "adm" and AccountName !startswith "sys"
| join DeviceInfo on DeviceName
| project AccountName, DeviceName, LoggedOnUsers, DeviceId, ReportId, Timestamp
| summarize count() by AccountName, DeviceName, LoggedOnUsers, DeviceId, ReportId, Timestamp
